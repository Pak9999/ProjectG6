



chosen_continent = forms.ChoiceField [continent_name from continent + NULL]

if chosen_continent is NULL:
    pass

elif chosen_continent:
    chosen_country = forms.ChoiceField [country_name from country where country(continent_id) = continent(continent_id) + NULL]

    if chosen_country is NULL
        write new country_article

    elif chosen_country:

        chosen_region = forms.ChoiceField [region_name from region where region(country_id) = country(country_id) + NULL]

        if chosen_region is NULL:
            write new region_article

        elif chosen_region:

            in_region_choice = forms.ChoiceField ['city', 'poi',]

            if in_region_choice = 'city'

                chosen_city = forms.ChoiceField [city_name from city where city(region_id) = region(region_id) + NULL]


                if chosen_city is NULL
                    write new city_article

                elif chosen_city:
                    write new poi_article
            
            elif in_region_choice = 'poi'
                write new poi_article

            else:
                ERROR_PLACEHOLDER

        else:
            ERROR_PLACEHOLDER
    else:
        ERROR_PLACEHOLDER

else:
    ERROR_PLACEHOLDER
